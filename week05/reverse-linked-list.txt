class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        if(head == NULL)return head;
        
        
        auto current = head;
        ListNode* previous = NULL;
        
        while(current){
            auto next = current->next;
             current->next = previous;
             previous = current;
             current  = next; 
        }
        
        return previous;
    }
};
